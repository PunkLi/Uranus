67  void readData(std::string& file_name,
68  DataSet& buffer,
69  int feature_num,
70  int sample_num )
71  {
72  buffer.resize(sample_num);
73  std::ifstream fin(file_name);
74  std::string line;
75 
76  for (int index = 0; index < sample_num; ++index)
77  {
78  std::getline(fin, line);
79  std::istringstream sin(line);
80  std::vector<std::string> fields;
81  std::string field;
82  while (std::getline(sin, field, ','))
83  fields.push_back(field);
84 
85  int idx = feature_num;
86  std::vector<double> feature_vec(fields.size());
87  for (int idx = 0; idx < feature_num; ++idx)
88  feature_vec[idx] = std::stod(Trim(fields[idx]));
89  std::string label = Trim(fields[idx]);
90 
91  buffer[index] = std::make_pair(feature_vec, label);
92 
93  std::cout << "处理之后的字符串："
94  << feature_vec[0] << "\t"
95  << feature_vec[1] << "\t"
96  << feature_vec[2] << "\t"
97  << feature_vec[3] << "\t"
98  << std::endl;
99  }
100  }

107  std::string Trim(std::string& str)
108  {
109  str.erase(0, str.find_first_not_of(" \t\r\n"));
110  str.erase(str.find_last_not_of(" \t\r\n") + 1);
111  return str;
112  }
